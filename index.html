chart = {
  const width = 800;
  const height = 600;
  
  const drag = simulation => {
    function started(event) {
      if (!event.active) simulation.alphaTarget(0.3).restart();
      event.subject.fx = event.subject.x;
      event.subject.fy = event.subject.y;
    }
    function dragged(event) {
      event.subject.fx = event.x;
      event.subject.fy = event.y;
    }
    function ended(event) {
      if (!event.active) simulation.alphaTarget(0);
      event.subject.fx = null;
      event.subject.fy = null;
    }
    return d3.drag().on("start", started).on("drag", dragged).on("end", ended);
  };

  const svg = d3.create("svg")
      .attr("viewBox", [0, 0, width, height])
      .style("max-width", "100%")
      .style("height", "auto");

  const simulation = d3.forceSimulation(graphData.nodes)
      .force("link", d3.forceLink(graphData.links).id(d => d.id).distance(60))
      .force("charge", d3.forceManyBody().strength(-100))
      .force("center", d3.forceCenter(width / 2, height / 2));

  const link = svg.append("g")
      .attr("stroke", "#e2e2e2")
      .attr("stroke-opacity", 0.6)
    .selectAll("line")
    .data(graphData.links)
    .join("line")
      .attr("stroke-width", 1.5);

  const node = svg.append("g")
      .attr("stroke", "#fff")
      .attr("stroke-width", 1.5)
    .selectAll("circle")
    .data(graphData.nodes)
    .join("circle")
      .attr("r", d => d.radius)
      .attr("fill", d => colors(d.group))
      .call(drag(simulation));

  const label = svg.append("g")
      .style("font-family", "sans-serif")
      .style("font-size", "11px")
      .style("pointer-events", "none")
    .selectAll("text")
    .data(graphData.nodes)
    .join("text")
      .attr("class", "node-label")
      .text(d => d.id)
      .attr("dx", d => d.radius + 5)
      .attr("dy", 4);

  simulation.on("tick", () => {
    link.attr("x1", d => d.source.x).attr("y1", d => d.source.y)
        .attr("x2", d => d.target.x).attr("y2", d => d.target.y);
    node.attr("cx", d => d.x).attr("cy", d => d.y);
    label.attr("x", d => d.x).attr("y", d => d.y);
  });

//text color swap hack
const themeObserver = new MutationObserver(() => {
  const isDark = document.documentElement.getAttribute("data-bs-theme") === "dark"
    || document.body.classList.contains("quarto-dark");
  label.style("fill", isDark ? "#eeeeee" : "#000000");
});

themeObserver.observe(document.documentElement, { attributes: true });
themeObserver.observe(document.body, { attributes: true, attributeFilter: ["class"] });

// Set initial color on load
const isDarkOnLoad = document.body.classList.contains("quarto-dark");
label.style("fill", isDarkOnLoad ? "#eeeeee" : "#000000");

//



  return svg.node();
}

I completed my PhD in Technology at Purdue University where I completed research on video game players and spatial ability. I developed a novel taxonomy for characterizing video games based on visual-spatial features and then had several hundred respondents describe their most-played games and complete several spatial ability tests. I used machine learning analysis techniques to group respondents by game type and then analyzed their spatial ability performance by group and gender.

I completed my second Master’s degree in Technology at Purdue University where I researched interactive computer graphics, virtual reality, and learning in dynamic virtual environments. I created a novel VR environment and explored how learning could take place withing such an environment.

I completed my first Master’s degree in Horticulture at Purdue University where I studied the genetic response of plants to touch. My model system was soybeans that I grew in a dark room to isolate non-photosynthetic development.

I completed my Bachelor’s degree in Biology at Henderson State University and completed independent genetics research on aquatic invertebrates.
